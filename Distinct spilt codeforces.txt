void solve()
{  
    int n;
    cin>>n;
    string s;
    cin>>s;
    vector<int>suffix(n,0),prefix(n,0);
    set<char>st;
    for(int i=0;i<n;i++){
        st.insert(s[i]);
        prefix[i]=st.size();
    }
    st.clear();
    for(int i=n-1;i>=0;i--){
        st.insert(s[i]);
        suffix[i]=st.size();
    }
    int sum=0,ans=0;
    for(int i=0;i<n-1;i++){
        sum=prefix[i]+suffix[i+1];
        ans=max(ans,sum);
    }
    cout<<ans<<endl;    
}

    while (t--)
    solve();
